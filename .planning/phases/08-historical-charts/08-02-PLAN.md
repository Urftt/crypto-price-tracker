#### Step 2: Add candle endpoint inside create_app()

Add this endpoint inside `create_app()`, after the existing alert endpoints and before the `@app.get("/")` index route:

```python
# --- Candle/Chart endpoints ---

@app.get("/api/candles/{symbol}")
def api_candles(
    symbol: str,
    interval: str = Query(default="4h", pattern="^(1m|5m|15m|30m|1h|2h|4h|6h|8h|12h|1d)$"),
    limit: int = Query(default=42, ge=1, le=1440),
):
    """Return OHLCV candle data for a market in chronological order."""
    symbol = symbol.upper()
    market = f"{symbol}-EUR"
    try:
        candles = get_candles(market, interval=interval, limit=limit)
    except httpx.HTTPStatusError as e:
        if e.response.status_code == 400:
            raise HTTPException(
                status_code=404,
                detail=f"No candle data for '{symbol}'",
            )
        raise
    return [dataclasses.asdict(c) for c in candles]
```

**Important notes:**
- The `pattern` parameter on `interval` validates against all Bitvavo-supported intervals.
- `limit` is constrained to 1-1440 (reasonable upper bound).
- Bitvavo returns HTTP 400 for invalid markets, which we translate to 404 for the client.
- Response is a plain JSON array of candle objects (each with timestamp, open, high, low, close, volume).
- The `get_candles()` convenience function from `api.py` handles client lifecycle and data reversal.

```javascript
/* --- Chart in modal --- */
let currentChartSymbol = '';

async function loadChart(symbol, interval, limit) {
  currentChartSymbol = symbol;
  const chartSection = document.getElementById('chart-section');
  const container = document.getElementById('chart-container');

  try {
    const resp = await fetch(`/api/candles/${symbol}?interval=${interval}&limit=${limit}`);
    if (!resp.ok) {
      chartSection.style.display = 'none';
      return;
    }
    const candles = await resp.json();
    if (candles.length === 0) {
      chartSection.style.display = 'none';
      return;
    }

    chartSection.style.display = '';

    const trace = {
      x: candles.map(c => new Date(c.timestamp)),
      y: candles.map(c => c.close),
      type: 'scatter',
      mode: 'lines',
      line: { color: '#58a6ff', width: 2 },
      hovertemplate: '%{x|%b %d, %H:%M}<br>\u20ac %{y:,.2f}<extra></extra>'
    };

    const layout = {
      paper_bgcolor: '#161b22',
      plot_bgcolor: '#161b22',
      font: { color: '#c9d1d9', family: 'monospace' },
      xaxis: {
        gridcolor: '#21262d',
        tickformat: interval === '1d' ? '%b %d' : '%b %d %H:%M'
      },
      yaxis: {
        gridcolor: '#21262d',
        tickprefix: '\u20ac ',
        tickformat: ',.2f'
      },
      margin: { t: 10, r: 20, b: 40, l: 70 },
      height: 250
    };

    const config = { responsive: true, displayModeBar: false };
    Plotly.newPlot(container, [trace], layout, config);
  } catch (err) {
    chartSection.style.display = 'none';
  }
}

function switchPeriod(period) {
  // Update button states
  document.querySelectorAll('.chart-period-btn').forEach(btn => btn.classList.remove('active'));
  const activeBtn = period === '7d'
    ? document.querySelector('.chart-period-btn:first-child')
    : document.querySelector('.chart-period-btn:last-child');
  if (activeBtn) activeBtn.classList.add('active');

  // Load chart with appropriate interval and limit
  if (period === '7d') {
    loadChart(currentChartSymbol, '4h', 42);
  } else {
    loadChart(currentChartSymbol, '1d', 30);
  }
}
```
