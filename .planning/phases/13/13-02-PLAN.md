### Step 1: Create useOffline hook
**File:** `frontend/src/hooks/useOffline.js`
**Action:** create
**Why:** Encapsulates offline detection using the browser's navigator.onLine API and online/offline events. Used by the offline banner and can also be consumed by other components.

Create `frontend/src/hooks/useOffline.js`:

```jsx
import { useState, useEffect } from 'react';

export function useOffline() {
  const [isOffline, setIsOffline] = useState(!navigator.onLine);

  useEffect(() => {
    const goOffline = () => setIsOffline(true);
    const goOnline = () => setIsOffline(false);

    window.addEventListener('offline', goOffline);
    window.addEventListener('online', goOnline);

    return () => {
      window.removeEventListener('offline', goOffline);
      window.removeEventListener('online', goOnline);
    };
  }, []);

  return isOffline;
}
```

Returns a boolean `isOffline` that reactively updates when the network status changes. Initializes from `navigator.onLine` on first render.

### Step 2: Create useInstallPrompt hook
**File:** `frontend/src/hooks/useInstallPrompt.js`
**Action:** create
**Why:** Captures the browser's `beforeinstallprompt` event and provides a programmatic install() function. Only Chrome/Edge fire this event; Safari uses its own share-sheet flow.

Create `frontend/src/hooks/useInstallPrompt.js`:

```jsx
import { useState, useEffect } from 'react';

export function useInstallPrompt() {
  const [deferredPrompt, setDeferredPrompt] = useState(null);
  const [isInstallable, setIsInstallable] = useState(false);

  useEffect(() => {
    const handler = (e) => {
      e.preventDefault();
      setDeferredPrompt(e);
      setIsInstallable(true);
    };
    window.addEventListener('beforeinstallprompt', handler);

    // Hide install button if app is already installed
    window.addEventListener('appinstalled', () => {
      setDeferredPrompt(null);
      setIsInstallable(false);
    });

    return () => {
      window.removeEventListener('beforeinstallprompt', handler);
    };
  }, []);

  const install = async () => {
    if (!deferredPrompt) return;
    deferredPrompt.prompt();
    const { outcome } = await deferredPrompt.userChoice;
    setDeferredPrompt(null);
    setIsInstallable(false);
    return outcome;
  };

  return { isInstallable, install };
}
```

Key behaviors:
- `isInstallable` is `true` only when the browser has fired `beforeinstallprompt` AND the app is not yet installed
- `install()` triggers the native browser install dialog and returns the user's choice ('accepted' or 'dismissed')
- The `appinstalled` listener hides the button after successful installation
- On browsers that don't support `beforeinstallprompt` (Safari, Firefox), `isInstallable` stays `false` — the button never renders

### Step 3: Create OfflineBanner component
**File:** `frontend/src/components/OfflineBanner.jsx`
**Action:** create
**Why:** Shows a persistent banner at the top of the page when the user is offline, informing them that they're seeing cached data.

Create `frontend/src/components/OfflineBanner.jsx`:

```jsx
import { useOffline } from '../hooks/useOffline';

function OfflineBanner() {
  const isOffline = useOffline();

  if (!isOffline) return null;

  return (
    <div className="bg-down/20 border border-down/40 text-down text-sm text-center py-2 px-4">
      You're offline — showing last known data
    </div>
  );
}

export default OfflineBanner;
```

Styling uses the project's existing Tailwind theme:
- `bg-down/20` — translucent red background (#f85149 at 20% opacity)
- `border-down/40` — red border at 40% opacity
- `text-down` — red text matching the project's "negative" color
- Centered text with small padding — unobtrusive but visible

### Step 4: Create InstallButton component

**Why:** Provides a button in the header that triggers the PWA install prompt. Only renders when the browser supports installation (Chrome/Edge).

function InstallButton() {

  return (

    >
      Install App
    </button>
  );
}

export default InstallButton;
```

Styling uses the project's accent color (#58a6ff) with translucent background:

# ---- PWA static file tests ----


def test_manifest_served(client):
    """GET /manifest.json should return the PWA manifest."""
    response = client.get("/manifest.json")
    assert response.status_code == 200
    data = response.json()
    assert data["name"] == "Crypto Price Tracker"
    assert data["display"] == "standalone"
    assert data["start_url"] == "/"
    assert len(data["icons"]) >= 2


def test_service_worker_served(client):
    """GET /sw.js should return the service worker script."""
    response = client.get("/sw.js")
    assert response.status_code == 200
    assert "text/javascript" in response.headers.get("content-type", "") or response.status_code == 200


def test_pwa_icons_served(client):
    """PWA icon files should be accessible."""
    for icon_path in ["/pwa-192x192.png", "/pwa-512x512.png", "/apple-touch-icon-180x180.png"]:
        response = client.get(icon_path)
        assert response.status_code == 200, f"Icon {icon_path} not served"


def test_favicon_served(client):
    """GET /favicon.ico should return the favicon."""
    response = client.get("/favicon.ico")
    assert response.status_code == 200


def test_index_html_contains_manifest_link(client):
    """The built index.html should contain a manifest link tag."""
    response = client.get("/")
    assert response.status_code == 200
    content = response.text
    assert "manifest" in content


def test_index_html_contains_theme_color(client):
    """The built index.html should contain the theme-color meta tag."""
    response = client.get("/")
    assert response.status_code == 200
    content = response.text
    assert "theme-color" in content
```

These tests verify:
- The manifest.json is served with correct JSON content
- The service worker JavaScript file is accessible
- All PWA icon files are served through the SPA catch-all route
- The built index.html contains the auto-injected manifest link and manually-added theme-color meta tag
